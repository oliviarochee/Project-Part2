DROP TABLE "USERDATA";

CREATE TABLE "USERDATA"
(
"USER_ID" INT not null primary key
        GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),   
"EMAIL" VARCHAR(50), 
"PASSWORD" VARCHAR(50), 
"FNAME" VARCHAR(50),   
"LNAME" VARCHAR(50),
"USERTYPE" VARCHAR(10)
);

DROP TABLE "PRODUCT";

CREATE TABLE "PRODUCT"
(
"PRODUCT_ID" INT not null primary key
        GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),   
"CODE" VARCHAR(50),
"NAME" VARCHAR(50),
"TYPE" VARCHAR(4),
"IMAGE" VARCHAR(200),
"LONG_DESCRIPTION" VARCHAR(100),
"COLOR" VARCHAR(50),
"UNITCOST" VARCHAR(50)
);
